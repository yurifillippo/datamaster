#Carregar dados no hive - clientes

USE banco_mock;

CREATE TABLE clientes (
    id INT,
    nome STRING,
    sobrenome STRING,
    data_nascimento DATE,
    email STRING,
    telefone STRING,
    endereco STRING,
    cidade STRING,
    estado STRING,
    cep STRING,
    data_criacao TIMESTAMP
)
STORED AS PARQUET;

LOAD DATA LOCAL INPATH 'path/to/clientes_data.parquet' INTO TABLE clientes_parquet;



USE produtos_mock;

CREATE TABLE produtos_parquet (
    id INT,
    nome STRING,
    categoria STRING,
    descricao STRING,
    preco DECIMAL(10, 2),
    quantidade_estoque INT,
    data_adicao TIMESTAMP
)
STORED AS PARQUET;

LOAD DATA LOCAL INPATH 'path/to/produtos_data.parquet' INTO TABLE produtos_parquet;



USE produtos_mock;

CREATE TABLE clientes_produtos_parquet (
    produto_id INT,
    cliente_id INT,
    produto_nome STRING,
    quantidade INT,
    data_compra TIMESTAMP
)
STORED AS PARQUET;

LOAD DATA LOCAL INPATH 'path/to/clientes_produtos_data.parquet' INTO TABLE clientes_produtos_parquet;
